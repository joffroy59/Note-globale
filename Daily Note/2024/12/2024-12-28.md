```dataview
TABLE WITHOUT ID
  file.link as Date,
  choice(IF_hrs > 18, "✅", "❌") as IF,
  choice(pushups > 1, "✅", "❌") as pushups,
  reading as Reading
FROM "Daily Note"
WHERE file.link > (date(now) - dur(7 days))
SORT file.day DESC
```
